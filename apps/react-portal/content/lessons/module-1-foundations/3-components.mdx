## Props: passing data into components

<AnnotatedCode
  title="UserCard.tsx"
  language="tsx"
  code={`interface UserCardProps {
  name: string;
  role: string;
  avatarUrl?: string;
}

function UserCard({ name, role, avatarUrl }: UserCardProps) {
  return (
    <div className="card">
      {avatarUrl && <img src={avatarUrl} alt={name} />}
      <h3>{name}</h3>
      <p>{role}</p>
    </div>
  );
}

// Usage
<UserCard name="Sudharsan" role="Architect" />`}
  annotations={[

    { line: 10, label: 'Conditional render', explanation: '{avatarUrl && <img />} renders the image only when avatarUrl has a value. This is the standard JSX pattern for conditional content.' },

<DangerZone
  title="Never modify props directly"
  wrong={`function BadComponent(props) {\n  props.name = 'Modified'; // This will throw in strict mode\n  return <p>{props.name}</p>;\n}`}
  right={`function GoodComponent({ name }) {\n  // Props are read-only — use local state if you need to transform them\n  return <p>{name}</p>;\n}`}
  explanation="Props flow one direction — from parent to child. A component that modifies its own props breaks the predictability that makes React easy to reason about. Use state for values that need to change."
/>
