## Building a custom hook

<AnnotatedCode
  title="useWindowSize.ts"
  language="tsx"
  code={`import { useState, useEffect } from 'react';

interface WindowSize {
  width: number;
  height: number;
}

export function useWindowSize(): WindowSize {
  const [size, setSize] = useState<WindowSize>({
    width: window.innerWidth,
    height: window.innerHeight,
  });

  useEffect(() => {
    function handleResize() {
      setSize({ width: window.innerWidth, height: window.innerHeight });
    }
    window.addEventListener('resize', handleResize);
    return () => window.removeEventListener('resize', handleResize);
  }, []);

  return size;
}

// Usage — any component can now get live window dimensions
function ResponsiveLayout() {
  const { width } = useWindowSize();
  return <p>Viewport: {width}px</p>;
}`}
  annotations={[
    { line: 8, label: 'use prefix is required', explanation: 'The name must start with "use" — this signals to React (and linters) that this function can call hooks. React enforces the Rules of Hooks based on this naming convention.' },
    { line: 14, label: 'Hooks inside hooks', explanation: 'Custom hooks can call useState, useEffect, and other custom hooks. The logic runs in the calling component — each component instance gets its own isolated copy of the state.' },
    { line: 22, label: 'Return values freely', explanation: 'Unlike components, hooks can return any value — a single value, an array, or an object. Design the return type to match how callers will use it.' },

<DangerZone
  title="Custom hooks must follow the Rules of Hooks"
  wrong={`// Never call hooks conditionally — even inside a custom hook\nfunction useBadHook(enabled: boolean) {\n  if (enabled) {\n    const [x, setX] = useState(0); // Rules violation: conditional hook call\n  }\n}`}
  right={`// Gate the logic, not the hook call\nfunction useGoodHook(enabled: boolean) {\n  const [x, setX] = useState(0); // Always called\n  useEffect(() => {\n    if (!enabled) return; // Early return inside effect is fine\n    // do something\n  }, [enabled]);\n  return x;\n}`}

<TryItInline
  defaultCode={`function useCounter(initial = 0) {
  const [count, setCount] = React.useState(initial);
  const increment = () => setCount(c => c + 1);
  const decrement = () => setCount(c => c - 1);
  const reset = () => setCount(initial);
  return { count, increment, decrement, reset };
}

function App() {
  const { count, increment, decrement, reset } = useCounter(0);
  return (
    <div style={{textAlign:'center', marginTop:'40px', fontFamily:'system-ui'}}>
      <h2 style={{fontSize:'2.5rem', margin:'0 0 16px'}}>{count}</h2>
      <button onClick={decrement} style={{padding:'8px 20px', fontSize:'1rem', marginRight:'8px'}}>−</button>
      <button onClick={reset} style={{padding:'8px 20px', fontSize:'1rem', marginRight:'8px'}}>Reset</button>
      <button onClick={increment} style={{padding:'8px 20px', fontSize:'1rem'}}>+</button>
    </div>
  );
}`}

<DeepDive title="The Rules of Hooks — why they exist">
  React tracks hook state using an array indexed by call order. If you put a hook inside an if block, the array index shifts on the next render and React reads the wrong state for every subsequent hook. The linter rule `react-hooks/rules-of-hooks` enforces this statically. Custom hooks are just functions — they inherit the rules because they call the same built-in hooks internally.
</DeepDive>
