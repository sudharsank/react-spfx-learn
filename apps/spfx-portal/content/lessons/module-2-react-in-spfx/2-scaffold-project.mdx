## Key files generated

After running `yo @microsoft/sharepoint` for a web part named `HelloWorld`, the structure looks like this:

```text
my-webpart/
├── config/
│   ├── heft.json               ← Heft build config
│   ├── package-solution.json   ← Solution metadata + CDN settings
│   └── serve.json              ← Local dev server config (legacy, ignored by heft start)
├── src/
│   └── webparts/
│       └── helloWorld/
│           ├── HelloWorldWebPart.ts        ← Web part class
│           ├── HelloWorldWebPart.manifest.json
│           └── components/
│               ├── HelloWorld.tsx          ← React component
│               └── HelloWorld.module.scss
├── package.json
└── tsconfig.json
```

## The web part class bootstraps React

<AnnotatedCode
  title="HelloWorldWebPart.ts — how SPFx bootstraps React"
  language="typescript"
  code={`import * as React from 'react';
import * as ReactDom from 'react-dom';
import { Version } from '@microsoft/sp-core-library';
import { BaseClientSideWebPart } from '@microsoft/sp-webpart-base';
import HelloWorld, { IHelloWorldProps } from './components/HelloWorld';

export default class HelloWorldWebPart extends BaseClientSideWebPart<{}> {
  public render(): void {
    const element: React.ReactElement<IHelloWorldProps> = React.createElement(
      HelloWorld,
      {
        description: this.properties.description,
        context: this.context,
      }
    );
    ReactDom.render(element, this.domElement);
  }

  protected onDispose(): void {
    ReactDom.unmountComponentAtNode(this.domElement);
  }

  protected get dataVersion(): Version {
    return Version.parse('1.0');
  }
}`}
  annotations={[
    { line: 7, label: 'BaseClientSideWebPart', explanation: 'This is the SPFx base class. It handles lifecycle events from SharePoint: when to render, when to re-render on property change, and when to clean up.' },
    { line: 8, label: 'render()', explanation: 'SharePoint calls this when it needs to display your web part. You create a React element and mount it into this.domElement — the div SharePoint gave you.' },
    { line: 19, label: 'onDispose()', explanation: 'SharePoint calls this when the web part is removed from the page. Unmounting the React tree prevents memory leaks from event listeners and subscriptions.' },
    { line: 23, label: 'dataVersion', explanation: 'This version number is attached to your web part properties. If you change the property schema, bump this version so SharePoint can migrate old data.' },
  ]}

## heft.json — the build config

  title="config/heft.json — minimal example"

  "phasesByName": {
    "build": {
      "phaseDescription": "Compiles TypeScript and copies assets",
      "tasksByName": {
        "typescript": {
          "taskPlugin": {
            "pluginPackage": "@rushstack/heft-typescript-plugin"
          }
        }
      }
    }
  }
}`}
